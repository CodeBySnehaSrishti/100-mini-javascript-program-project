Function to validate the form function validateForm() { const input = document.getElementById('username'); const errorMessage = document.getElementById('error-message'); // If the input is empty, show an error and add the shake effect if (input.value.trim() === '') { errorMessage.style.display = 'block'; input.classList.add('shake'); // Remove the shake class after the animation completes setTimeout(() => { input.classList.remove('shake'); }, 500); } else { errorMessage.style.display = 'none'; alert('Form submitted successfully!'); } } // Attach event listener to the button document.getElementById('submit-btn').addEventListener('click', validateForm);
